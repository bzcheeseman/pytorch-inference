68  inline af::array conv2d(const conv_params_t &params,
69  const af::array &input,
70  const af::array &filters,
71  const af::array &bias,
72  const bool &has_bias){
73  long Cin = input.dims(2); long Cout = filters.dims(3); long batch = input.dims(3);
74 
75  if (has_bias)
76  check_size(bias.dims(2), Cout, __func__);
77 
78  check_size(filters.dims(2), Cin, __func__);
79  check_size(filters.dims(0), params.filter_x, __func__);
80  check_size(filters.dims(1), params.filter_y, __func__);
81 
82  long h_in = input.dims(0); long w_in = input.dims(0);
83  long h_out = (int)floor((input.dims(0) - params.filter_x + 2*params.pad_x)/params.stride_x + 1);
84  long w_out = (int)floor((input.dims(1) - params.filter_y + 2*params.pad_y)/params.stride_y + 1);
85 
86  af::array out = af::constant(0, h_out, w_out, Cout, batch); // (h, w, k, n)
87 
88  af::array in = af::unwrap(input, params.filter_x, params.filter_y,
89  params.stride_x, params.stride_y,
90  params.pad_x, params.pad_y);
91  af::array f = af::reorder(filters, 3, 0, 1, 2);
92  f = af::moddims(f, Cout, params.filter_x*params.filter_y, Cin); // (Cout, h*w, Cin)
93  af::array b;
94  if (has_bias)
95  b = af::tile(bias, h_out, w_out, 1, 1);
96 
97  // input is (fx*fy, ox*oy, Cin, n)
98  // filters is (Cout, fx*fy, Cin)
99  // need to multiply each of Cout filters onto input
100  // so take Cout x fx*fy . fx*fy x ox*oy = Cout x ox*oy
101 
102  // Once batched matmul is available, gotta get rid of these for loops (or just replace with gfor?)
103 #pragma omp parallel for // parallelize the batch dimension, maybe we can replace in the future?
104  for (int i = 0; i < batch; i++){
105  for (int k = 0; k < Cin; k++){
106  out(af::span, af::span, af::span, i) +=
107  af::moddims(
108  af::reorder(
109  af::matmul(f(af::span, af::span, k, af::span), in(af::span, af::span, k, i)),
110  3, 1, 2, 0),
111  h_out, w_out, Cout);
112  }
113  if (has_bias)
114  out(af::span, af::span, af::span, i) += b;
115  }
116 
117  return out;
118 
119  }
