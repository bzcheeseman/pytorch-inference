36  inline std::vector<af::array> split_branch(const af::array &input, const int &slices, const int &dim){
37  check_size(input.dims(dim)%slices, 0, __func__); // it's gotta be an even split
38  check_num_leq(slices, 10, __func__); // we can only concatenate 10 items so don't split more than that
39  std::vector<af::array> out;
40  switch (dim) { // each array has a slice that is an equal piece of that dimension. Also af::seq is inclusive.
41  case 0 : // change to gfor?
42  for (int i = 0; i < input.dims(0); i += input.dims(0)/slices) {
43  out.push_back(input(af::seq(i, i+input.dims(0)/slices-1), af::span, af::span, af::span));
44  }
45  break;
46 
47  case 1 :
48  for (int i = 0; i < input.dims(1); i += input.dims(1)/slices) {
49  out.push_back(input(af::span, af::seq(i, i+input.dims(1)/slices-1), af::span, af::span));
50  }
51  break;
52 
53  case 2 :
54  for (int i = 0; i < input.dims(2); i += input.dims(2)/slices) {
55  out.push_back(input(af::span, af::span, af::seq(i, i+input.dims(2)/slices-1), af::span));
56  }
57  break;
58 
59  case 3 :
60  for (int i = 0; i < input.dims(3); i += input.dims(3)/slices) {
61  out.push_back(input(af::span, af::span, af::span, af::seq(i, i+input.dims(3)/slices-1)));
62  }
63  break;
64 
65  default: out.push_back(input); std::cerr << "Dimension not in range, no-op." << std::endl; break;
66  }
67  return out;
68  }
